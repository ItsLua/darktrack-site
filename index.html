            <i class="fa-solid fa-comment-dots text-xl"></i>

        </button>

    </div>


    <!-- ================= JAVASCRIPT LOGIC ================= -->

    <script>

        // State Management

        let currentUser = null;

        let isChatOpen = false;


        // DOM Elements

        const authView = document.getElementById('auth-view');

        const appView = document.getElementById('app-view');

        const chatWidget = document.getElementById('chat-widget');

        const loginForm = document.getElementById('login-form');

        const signupForm = document.getElementById('signup-form');

        const tabLogin = document.getElementById('tab-login');

        const tabSignup = document.getElementById('tab-signup');

        const chatWindow = document.getElementById('chat-window');

        const chatMessages = document.getElementById('chat-messages');

        const userDisplayName = document.getElementById('user-display-name');


        // Initialization

        function init() {

            // Check if user was previously logged in (Simple session storage simulation)

            const storedUser = localStorage.getItem('darktrack_user');

            if (storedUser) {

                loginUser(JSON.parse(storedUser));

            } else {

                showAuth();

            }

        }


        // --- Authentication Logic ---


        function switchAuthMode(mode) {

            if (mode === 'login') {

                loginForm.classList.remove('hidden');

                signupForm.classList.add('hidden');

                tabLogin.classList.add('border-brand-primary', 'text-white');

                tabLogin.classList.remove('border-transparent', 'text-brand-muted');

                tabSignup.classList.add('border-transparent', 'text-brand-muted');

                tabSignup.classList.remove('border-brand-primary', 'text-white');

            } else {

                loginForm.classList.add('hidden');

                signupForm.classList.remove('hidden');

                tabSignup.classList.add('border-brand-primary', 'text-white');

                tabSignup.classList.remove('border-transparent', 'text-brand-muted');

                tabLogin.classList.add('border-transparent', 'text-brand-muted');

                tabLogin.classList.remove('border-brand-primary', 'text-white');

            }

        }


        function handleLogin(e) {

            e.preventDefault();

            // In a real app, you'd fetch an API here.

            // For this demo, we simulate a successful login.

            const email = e.target.querySelector('input[type="email"]').value;

            const mockUser = { name: email.split('@')[0], email: email };

            

            // Simulate network delay

            const btn = e.target.querySelector('button');

            const originalText = btn.innerText;

            btn.innerText = "AUTHENTICATING...";

            btn.disabled = true;


            setTimeout(() => {

                loginUser(mockUser);

                btn.innerText = originalText;

                btn.disabled = false;

            }, 800);

        }


        function handleSignup(e) {

            e.preventDefault();

            const name = e.target.querySelector('input[type="text"]').value;

            const email = e.target.querySelector('input[type="email"]').value;

            const mockUser = { name: name, email: email };


             // Simulate network delay

             const btn = e.target.querySelector('button');

             const originalText = btn.innerText;

             btn.innerText = "ENCRYPTING DATA...";

             btn.disabled = true;


            setTimeout(() => {

                loginUser(mockUser);

                btn.innerText = originalText;

                btn.disabled = false;

            }, 800);

        }


        function loginUser(user) {

            currentUser = user;

            localStorage.setItem('darktrack_user', JSON.stringify(user));

            userDisplayName.textContent = user.name; // Update dashboard name

            

            // Animations to switch views

            authView.style.opacity = '0';

            setTimeout(() => {

                authView.classList.add('hidden');

                appView.classList.remove('hidden');

                chatWidget.classList.remove('hidden');

                // Trigger reflow/animation

                appView.classList.add('animate-fade-in');

            }, 300);

        }


        function handleLogout() {

            localStorage.removeItem('darktrack_user');

            currentUser = null;

            appView.classList.add('hidden');

            chatWidget.classList.add('hidden');

            authView.classList.remove('hidden');

            authView.style.opacity = '1';

            

            // Reset forms

            loginForm.reset();

            signupForm.reset();

        }


        function showAuth() {

            authView.classList.remove('hidden');

            appView.classList.add('hidden');

            chatWidget.classList.add('hidden');

        }


        // --- Chat Logic ---


        function toggleChat() {

            isChatOpen = !isChatOpen;

            if (isChatOpen) {

                chatWindow.classList.remove('hidden');

            } else {

                chatWindow.classList.add('hidden');

            }

        }


        function handleChatSubmit(e) {

            e.preventDefault();

            const input = document.getElementById('chat-input');

            const message = input.value.trim();

            if (!message) return;


            // Add User Message

            addMessage(message, 'user');

            input.value = '';


            // Simulate Bot Response

            setTimeout(() => {

                const responses = [

                    "Accessing database...",

                    "Protocol requires higher clearance for that query.",

                    "Confirmed. We are tracking that data point.",

                    "An agent will intercept your request shortly.",

                    "System nominal. How else can I assist?"

                ];

                const randomResponse = responses[Math.floor(Math.random() * responses.length)];

                addMessage(randomResponse, 'bot');

            }, 1000);

        }


        function addMessage(text, sender) {

            const div = document.createElement('div');

            div.className = "flex items-start chat-message";

            

            if (sender === 'bot') {

                div.innerHTML = `

                    <div class="w-8 h-8 rounded bg-brand-dark border border-brand-border flex items-center justify-center mr-2 flex-shrink-0">

                        <i class="fa-solid fa-robot text-brand-primary text-xs"></i>

                    </div>

                    <div class="bg-brand-dark border border-brand-border p-3 rounded-lg rounded-tl-none text-brand-muted">

                        ${text}

                    </div>

                `;

            } else {

                div.className += " justify-end";

                div.innerHTML = `

                    <div class="bg-brand-primary text-white p-3 rounded-lg rounded-tr-none">

                        ${text}

                    </div>

                `;

            }

            chatMessages.appendChild(div);

            chatMessages.scrollTop = chatMessages.scrollHeight;

        }


        // Start

        window.addEventListener('DOMContentLoaded', init);


    </script>

</body>

</html>
